#include <stdio.h>  // For standard input/output functions
#include <stdlib.h> // For rand and srand functions
#include <string.h> // For strcmp function
#include <ctype.h>  // For toupper function
#include <math.h>   // For fmod function
#include <time.h>   // For time function

#define MAX_QUESTIONS 10 

char questions[MAX_QUESTIONS][500] = {
    "What is the correct syntax to declare a pointer to an integer in C?", 
    "What will be the value of x after the following operation? x = 10 / 4;", 
    "What will be the output of printf(\"%d\", sizeof(int)) on a 32-bit system?",
    "Which loop executes at least once, regardless of the condition?", 
    "What is the purpose of the malloc() function in C?",
    "Which of the following correctly declares an array of 10 integers in C?",
    "What will be the output of the following code?\n\nint x = 5;\nif (x > 3)\n\tprintf(\"Yes\");\nelse\n\tprintf(\"No\");",
    "Which of the following functions is used to read a string in C?",
    "Which header file is required for using the printf() function in C?",
    "What is the default return type of the main() function in C?"
};

char choices[MAX_QUESTIONS][4][500] = {
    {"A. int ptr;\nB. int *ptr;\nC. pointer int ptr;\nD. int &ptr;"}, 
    {"A. 2.5\nB. 2\nC. 3\nD. 4"}, 
    {"A. 2\nB. 4\nC. 8\nD. 16"},
    {"A. for loop\nB. while loop\nC. do-while loop\nD. None of the above"}, 
    {"A. Allocates memory and initializes to zero\nB. Allocates memory dynamically\nC. Deallocates memory\nD. Copies memory"},
    {"A. int arr[10];\nB. array arr[10];\nC. int arr = 10;\nD. arr[10] int;"},
    {"A. No\nB. Yes\nC. Compilation error\nD. Undefined"},
    {"A. scanf()\nB. gets()\nC. fgets()\nD. Both B and C"},
    {"A. stdio.h\nB. conio.h\nC. string.h\nD. math.h"},
    {"A. void\nB. int\nC. float\nD. char"}
};

char correctAnswers[MAX_QUESTIONS] = {'B', 'B', 'B', 'C', 'B', 'A', 'B', 'D', 'A', 'B'};

void Quiz(int totalQuestions) {
    char answer;
    time_t startTime, endTime, currentTime;
    int correct = 0;
    int wrong = 0;
    const int timeLimit = 10 * 60;
    
    printf("\nStarting the quiz!\n");
    time(&startTime);
    
    for (int i = 0; i < totalQuestions; i++) {
        int attempts = 0;
        int isCorrect = 0;
        
        while (attempts < 2 && !isCorrect) {
            time(&currentTime);
            if (difftime(currentTime, startTime) >= timeLimit) {
                printf("\033[1;31mTime's up! Quiz terminated.\033[0m\n");
                goto endQuiz;
            }

            printf("\n\033[1;37mQuestion %d: %s\033[0m\n", i + 1, questions[i]);
            printf("=====================================================================\n");
            for (int j = 0; j < 3; j++) {
                printf("%s\n", choices[i][j]);
            }

            printf("Your answer: ");
            scanf_s(" %c", &answer, 1);
            
            if (toupper(answer) == correctAnswers[i]) {
                printf("\033[1;32mCorrect! You may proceed.\033[0m\n");
                correct++;
                isCorrect = 1;
            } else {
                attempts++;
                if (attempts < 2) {
                    printf("\033[1;31mWrong answer. Try again!\033[0m\n");
                } else {
                    printf("\033[1;31mWrong answer. Moving to the next question.\033[0m\n");
                    printf("\033[1;33mThe correct answer was: %c\033[0m\n", correctAnswers[i]);
                    wrong++;
                }
            }
        }
    }
    
endQuiz:
    time(&endTime);
    double timeTaken = difftime(endTime, startTime);
    int minutes = (int)(timeTaken / 60);
    double seconds = fmod(timeTaken, 60);
    
    printf("\n\033[1;36mQuiz Completed!\033[0m\n");
    printf("\033[1;33mTime Taken: %d minutes %.2f seconds\n\033[0m", minutes, seconds);
    printf("\033[1;37mCorrect Answers:\033[1;32m %d\n\033[0m", correct);
    printf("\033[1;37mWrong Answers:\033[1;31m %d\n\033[0m", wrong);
    printf("\033[1;37mCongratulations on completing the quiz!\n\033[0m");
}

int main() {
    char password[20];
    const char correctPassword[] = "bapple";
    
    do {
        printf("Enter the checkpoint password: ");
        scanf_s("%19s", password, (unsigned)_countof(password));
        
        if (strcmp(password, correctPassword) != 0) {
            printf("Invalid password! Try again.\n");
        }
    } while (strcmp(password, correctPassword) != 0);

    printf("\n");
    printf("Password accepted! Let's begin the quiz.\n");
    
    Quiz(10);
    
    return 0;
}
